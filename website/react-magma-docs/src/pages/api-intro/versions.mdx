import {
  Table,
  TableHead,
  TableRow,
  TableHeaderCell,
  TableBody,
  TableCell,
} from 'react-magma-dom';
import { getVersions } from './getVersions';

export const NextVersionTable = () => {
  const [tags, setTags] = React.useState([]);
  const [row, setRow] = React.useState();
  React.useEffect(() => {
    const getVersionsAndTags = new Promise((res) => res(getVersions()));
    getVersionsAndTags.then(function(val) {
      setTags(val.tags);
    });
  }, []);
  React.useEffect(() => {
    if (tags && tags.next) {
      setRow(<TableRow><TableCell>{tags.next.version}</TableCell><TableCell>{tags.next.date}</TableCell><TableCell><a href='https://next--upbeat-sinoussi-f675aa.netlify.app/'>Docs</a></TableCell></TableRow>);
    } else {
      setRow(<></>);
    }
  }, [tags]);
  return (
    <>
      <Table>
        <TableHead>
          <TableRow>
            <TableHeaderCell style={{width: '300px'}}>Version</TableHeaderCell>
            <TableHeaderCell>Release date</TableHeaderCell>
            <TableHeaderCell>Docs link</TableHeaderCell>
          </TableRow>
        </TableHead>
        <TableBody>
          {row}
        </TableBody>
      </Table>
    </>
  );
};

export const ReleasedVersionsTable = () => {
  const [allVersions, setAllVersions] = React.useState([]);
  const [rows, setRows] = React.useState();
  React.useEffect(() => {
    const getVersionsAndTags = new Promise((res) => res(getVersions()));
    getVersionsAndTags.then(function(val) {
      setAllVersions(val.versions);
    });
  }, []);
  React.useEffect(() => {
    if (allVersions) {
      const versions = allVersions.reverse();
      const tableRow = versions.map((row, i) => {
        return (
          <TableRow key={`row${i}`}>
            <TableCell key={`row${i}-version`}>{row.version}{i === 0 ? <span>&nbsp;✅&nbsp;</span> : ''}</TableCell>
             <TableCell key={`row${i}-date`}>{row.date}</TableCell>
             <TableCell key={`row${i}-docs`}><a href={`https://react-magma.cengage.com/version/${row.version}`}>Docs</a></TableCell>
          </TableRow>
        )});
      setRows(tableRow);
    } else {
      setRows(<></>);
    }
  }, [allVersions]);
  return(
    <>
      <Table>
          <TableHead>
            <TableRow>
              <TableHeaderCell style={{width: '300px'}}>Version</TableHeaderCell>
              <TableHeaderCell>Release date</TableHeaderCell>
              <TableHeaderCell>Docs link</TableHeaderCell>
            </TableRow>
          </TableHead>
          <TableBody>
            {rows}
          </TableBody>
        </Table>
    </>
  );
};
