import {
  Table,
  TableHead,
  TableRow,
  TableHeaderCell,
  TableBody,
  TableCell,
} from 'react-magma-dom';
import { getVersions } from './getVersions';


export const Thing = () => {
  const [result, setResult] = React.useState([]);
  React.useEffect(() => {
    const getVersionsAndTags = new Promise((res) => res(getVersions()));
    getVersionsAndTags.then(function(val) {
      console.log('---', val);
      setResult(val);
    });
  }, []);
  const getRowsData = () => {
    if (result && result.versions) {
      const versions = result.versions;
      const x = versions.map((item, key) => {
        return [item.version, item.time, `https://react-magma.cengage.com/version/${item.version}`];
      });
      console.log('+++++', x);
    } else {
      console.log('EMPTY');
      return 'empty';
    }
  };
  return(
    <>
      Results -- <br/>
      {getRowsData()}
    </>
  );
};

export const ReleasedVersionsTable = () => {
  const rows = [];
  
  return (
    <>
      <Table>
          <TableHead>
            <TableRow>
              <TableHeaderCell>Version</TableHeaderCell>
              <TableHeaderCell>Release date</TableHeaderCell>
              <TableHeaderCell>Docs link</TableHeaderCell>
            </TableRow>
          </TableHead>
          <TableBody>
            {rows.map((row, i) => (
              <TableRow key={`row${i}`}>
                {row.map((cell, j) => {
                    if (j === 2) {
                        return <TableCell key={`cell${i}_${j}`}><a href={cell}>Docs</a></TableCell>
                    }
                    return <TableCell key={`cell${i}_${j}`}>{cell}</TableCell>
                })}
              </TableRow>
            ))}
          </TableBody>
        </Table>
    </>
  );
};
