# Page Templates (work in progress)

Get started on your project more quickly with these ready-made page templates.

## Full-width Template

This template is ideal for applications that need as much of the screen as possible.

<figure>
  <img src="../../images/layout/template-full-width.png" alt=" " />
</figure>

```tsx
import React from 'react';
import styled from '@emotion/styled';
import { Header } from '@cengage-patterns/header';
import {
  Drawer,
  DrawerPosition,
  Grid,
  GridAlignContent,
  GridItem,
  GridJustifyContent,
  magma,
  useMediaQuery,
} from 'react-magma-dom';

const Container = styled(Grid)`
  grid-template-columns: repeat(4, 1fr);
  grid-template-rows: 0.2fr 1.5fr 1.2fr 0.8fr;
  grid-template-areas:
    'header header header header'
    'main main main main'
    'card1 card1 card2 card3'
    'footer footer footer footer';
  gap: 0.5em;
`;

export function Example() {
  const [showDrawer, setShowDrawer] = React.useState(false);

  const isSmallerScreen = useMediaQuery(
    `(max-width:${magma.breakpoints.small}px)`
  );

  return (
    <Container GridJustifyContent={GridJustifyContent.center}>
      <GridItem gridColumn="1 / 5" as="header">
        <Header
          isInverse
          logo={<strong>LOGO</strong>}
          breakpoint={magma.breakpoints.large}
          isCompact={isSmallerScreen}
          onMenuButtonClick={() => setShowDrawer(true)}
        />
      </GridItem>

      <Drawer
        onClose={() => setShowDrawer(false)}
        isOpen={showDrawer}
        position={DrawerPosition.left}
      >
        Content
      </Drawer>

      <GridItem
        gridColumn=" 1 / 5"
        as="main"
        style={{
          backgroundColor: magma.colors.primary500,
          color: magma.colors.neutral100,
          textAlign: 'center',
          padding: magma.spaceScale.spacing05,
          width: 'calc(100% - 48px)',
          margin: '0 auto',
        }}
      >
        Main
      </GridItem>
      <GridItem
        gridColumn="1 / 3"
        as="card1"
        style={{
          backgroundColor: magma.colors.primary100,
          textAlign: 'center',
          padding: magma.spaceScale.spacing05,
          margin: '0 0 0 25px',
        }}
      >
        Content
      </GridItem>
      <GridItem
        gridColumn="3 / 4"
        as="card2"
        style={{
          backgroundColor: magma.colors.primary300,
          color: magma.colors.neutral100,
          textAlign: 'center',
          padding: magma.spaceScale.spacing05,
        }}
      >
        Content
      </GridItem>
      <GridItem
        gridColumn="4 /5"
        as="card3"
        style={{
          backgroundColor: magma.colors.primary400,
          color: magma.colors.neutral100,
          textAlign: 'center',
          padding: magma.spaceScale.spacing05,
          margin: '0 25px 0 0',
        }}
      >
        Content
      </GridItem>
      <GridItem
        gridColumn="1 / 5"
        as="footer"
        style={{
          backgroundColor: magma.colors.primary600,
          color: magma.colors.neutral100,
          textAlign: 'center',
          padding: magma.spaceScale.spacing05,
          width: 'calc(100% - 48px)',
          margin: '0 auto',
        }}
      >
        Footer
      </GridItem>
    </Container>
  );
}
```

## Restricted-width Template

This template is great for pages that focus more on reading content like marketing materials, articles, etc.

<figure>
  <img src="../../images/layout/template-restricted-width.png" alt=" " />
</figure>

```tsx
import React from 'react';
import styled from '@emotion/styled';
import { Header } from '@cengage-patterns/header';

  Drawer,
  DrawerPosition,
  Grid,
  GridItem,
  magma,
  useMediaQuery,
} from 'react-magma-dom';

const Container = styled(Grid)`
  grid-template-columns: auto 1fr;
  grid-template-rows: auto 1fr;
  grid-template-areas:
    'header header'
    'content content'
    'article1 article2';
  gap: 1em;
`;

export function Example() {
  const [showDrawer, setShowDrawer] = React.useState(false);

  const isSmallerScreen = useMediaQuery(
    `(max-width:${magma.breakpoints.small}px)`
  );

  return (
    <Container>
      <GridItem gridColumn="1 / 3" as="header">
        <Header
          isInverse
          logo={<strong>LOGO</strong>}
          breakpoint={magma.breakpoints.large}
          isCompact={isSmallerScreen}
          onMenuButtonClick={() => setShowDrawer(true)}
        />
      </GridItem>

      <Drawer
        position={DrawerPosition.left}
        onClose={() => setShowDrawer(false)}
        isOpen={showDrawer}
      >
        Sidebar
      </Drawer>

      <GridItem
        gridColumn="1 / 3"
        as="content"
        style={{

          backgroundColor: magma.colors.warning500,

        gridColumn="1 / 2"

        gridColumn="2 / 3"

          padding: magma.spaceScale.spacing05,
          marginRight: '75px',
        }}
      >
        article 2
      </GridItem>
```

## Global Header with Sidebar

This template includes a screen region for a left sidebar and includes all of the necessary responsive behavior.

<figure>
  <img src="../../images/layout/template-sidebar-app-header.png" alt=" " />
</figure>

```tsx
import React from 'react';
import styled from '@emotion/styled';
import { Header } from '@cengage-patterns/header';
import {
  Drawer,
  DrawerPosition,
  Grid,
  GridItem,
  magma,
  useMediaQuery,
} from 'react-magma-dom';

const Container = styled(Grid)`
  display: grid;
  grid-template-columns: auto 1fr;
  grid-template-rows: auto 1fr;
  grid-template-areas:
    'sidebar header'
    'sidebar content';
  gap: 0.25em;
`;

const StyledSidebar = styled.div`
  width: 240px;
  grid-area: sidebar;
  background: ${magma.colors.neutral300};
  padding: ${magma.spaceScale.spacing05};
  height: calc(100vh - 32px);
`;

const Sidebar = () => (
  <StyledSidebar style={{ background: magma.colors.secondary700 }}>
    Sidebar
  </StyledSidebar>
);

export function Example() {
  const [openDrawer, setOpenDrawer] = React.useState(false);

  const isLargeScreen = useMediaQuery(
    `(max-width:${magma.breakpoints.large}px)`
  );

  const isSmallerScreen = useMediaQuery(
    `(max-width:${magma.breakpoints.small}px)`
  );

  return (
    <Container>
      {isLargeScreen ? (
        <Drawer position={DrawerPosition.left} isOpen={openDrawer}>
          <Sidebar />
        </Drawer>
      ) : (
        <Sidebar />
      )}

      <GridItem as="header" gridColumn="2 / 4">
        <Header
          isInverse
          logo={<strong>LOGO</strong>}
          breakpoint={magma.breakpoints.large}
          isCompact={isSmallerScreen}
          onMenuButtonClick={() => {
            setOpenDrawer(!openDrawer);
          }}
        />
      </GridItem>

      <GridItem
        as="content"
        gridRow="2"
        gridColumn="2 / 4"
        style={{
          maxWidth: '1600px',
          width: '100%',
          justifySelf: 'center',
          background: magma.colors.secondary500,
          padding: magma.spaceScale.spacing05,
          height: 'calc(100vh - 122px)',
        }}
      >
        Content
      </GridItem>
    </Container>
  );
}
```

## Application Header with Sidebar

This template also includes the sidebar screen region, but the header extends all the way across the top of the page.

<figure>
  <img src="../../images/layout/template-sidebar-global-header.png" alt=" " />
</figure>

```tsx
import React from 'react';
import styled from '@emotion/styled';
import { Header } from '@cengage-patterns/header';
import {
  Drawer,
  DrawerPosition,
  Grid,
  GridItem,
  magma,
  useMediaQuery,
} from 'react-magma-dom';

const Container = styled(Grid)`
  grid-template-columns: auto 1fr;
  grid-template-rows: auto 1fr;
  grid-template-areas:
    'header header'
    'sidebar content'
    'sidebar footer';
  gap: 2px;
  justify-content: 'center';
`;

const StyledSidebar = styled.div`
  width: 240px;
  grid-area: sidebar;
  background: ${magma.colors.primary200};
  padding: ${magma.spaceScale.spacing05};
  height: calc(100vh - 55px);
`;

const StyledFooter = styled.div`
  grid-area: footer;
  max-width: 1600px;
  width: 100%;
  justify-self: center;
  background: ${magma.colors.primary700};
  color: ${magma.colors.neutral100};
  padding: ${magma.spaceScale.spacing05};
`;

const Sidebar = () => (
  <StyledSidebar style={{ backgroundColor: magma.colors.primary200 }}>
    Sidebar
  </StyledSidebar>
);

const Footer = () => (
  <StyledFooter
    style={{
      backgroundColor: magma.colors.primary700,
      color: magma.colors.neutral100,
    }}
  >
    Footer
  </StyledFooter>
);

export function Example() {
  const [openDrawer, setOpenDrawer] = React.useState(false);

  const isLargeScreen = useMediaQuery(
    `(max-width:${magma.breakpoints.large}px)`
  );

    `(max-width:${magma.breakpoints.small}px)`

  return (
    <Container>
      {isLargeScreen ? (

          <Sidebar />
        </Drawer>
      ) : (
        <Sidebar />
      )}
      <GridItem as="header" gridColumn="1 / 4">
        <Header
          isInverse
          logo={<strong>LOGO</strong>}
          breakpoint={magma.breakpoints.large}
          isCompact={isSmallerScreen}
          onMenuButtonClick={() => {
            setOpenDrawer(!openDrawer);
          }}
        />
      </GridItem>

      <GridItem
        as="content"
        gridColumn="2 / 4"
        gridRow="2"
        style={{
          maxWidth: '1600px',
          width: '100%',
          justifySelf: 'center',
          background: magma.colors.primary500,
          color: magma.colors.neutral100,
          padding: magma.spaceScale.spacing05,
          height: 'calc(100vh - 122px)',
        }}
      >
        Content
      </GridItem>
      <Footer />
    </Container>

```
