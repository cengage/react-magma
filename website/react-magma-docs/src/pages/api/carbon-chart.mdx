## Basic Usage

```typescript
import React from 'react';
import { CarbonChart } from 'react-magma-dom';

export function Example() {
return <CarbonChart>CarbonChart content</CarbonChart>;
}
```

## InverseContext

The `isInverse` prop will render a dark background and light text. The children will
inherit the `isInverse` prop unless specified otherwise.

```typescript
import React from 'react';
import { CarbonChart } from 'react-magma-dom';

export function Example() {
return <CarbonChart isInverse>CarbonChart content</CarbonChart>;
}
```
