<LeadParagraph>
  AIButton is a customizable button component with animated gradients, icon
  support, and flexible sizing. It is designed for modern interfaces and
  supports both text and icon-only usage.
</LeadParagraph>

## Basic Usage

```tsx
import React from 'react';

import { AIButton } from 'react-magma-dom';

export function Example() {
  return <AIButton aria-label="AI Trigger">AI Trigger</AIButton>;
}
```

## Icon Only Button

For icon-only buttons, you must provide an `aria-label` for accessibility.

```tsx
import React from 'react';

import { AIButton } from 'react-magma-dom';

export function Example() {
  return <AIButton leadingIcon aria-label="AI Action" />;
}
```

## Leading and Trailing Icons

You can add icons to the left or right of the button text using `leadingIcon` and `trailingIcon` props.

```tsx
import React from 'react';

import { AIButton, ButtonGroup } from 'react-magma-dom';
import { SettingsIcon } from 'react-magma-icons';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton aria-label="Leading custom icon" leadingIcon={<SettingsIcon />}>
        Leading custom icon
      </AIButton>
      <AIButton leadingIcon={false} trailingIcon={<SettingsIcon />}>
        Trailing custom icon
      </AIButton>
      <AIButton leadingIcon={<SettingsIcon />} trailingIcon={<SettingsIcon />}>
        Both Icons
      </AIButton>
    </ButtonGroup>
  );
}
```

## Sizes

AIButton supports `small`, `medium` (default), and `large` sizes.

```tsx
import React from 'react';

import { AIButton, AIButtonSize, ButtonGroup } from 'react-magma-dom';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton aria-label="Small" size={AIButtonSize.small}>
        Small
      </AIButton>
      <AIButton aria-label="Medium" size={AIButtonSize.medium}>
        Medium
      </AIButton>
      <AIButton aria-label="Large" size={AIButtonSize.large}>
        Large
      </AIButton>
    </ButtonGroup>
  );
}
```

## Variants

There are two gradient variants: `variantA` (default) and `variantB`.

```tsx
import React from 'react';

import { AIButton, AIButtonVariant, ButtonGroup } from 'react-magma-dom';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton aria-label="Variant A" variant={AIButtonVariant.variantA}>
        Variant A
      </AIButton>
      <AIButton aria-label="Variant B" variant={AIButtonVariant.variantB}>
        Variant B
      </AIButton>
    </ButtonGroup>
  );
}
```

You can override gradient colors using `leftColor`, `rightColor`, `hoverColor`, and `pressedColor`.

```tsx
import React from 'react';

import { AIButton } from 'react-magma-dom';

export function Example() {
  return (
    <AIButton
      aria-label="Custom Colors"
      leftColor="#00f"
      rightColor="#0ff"
      hoverColor="#0f0"
      pressedColor="#f00"
    >
      Custom Colors
    </AIButton>
  );
}
```

## Shapes

Shapes include `fill` (default), `leftCap`, `rightCap`, and `round` (used for icon-only buttons).

```tsx
import React from 'react';

import { AIButton, AIButtonShape, ButtonGroup } from 'react-magma-dom';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton aria-label="Fill" shape={AIButtonShape.fill}>
        Fill
      </AIButton>
      <AIButton aria-label="Left Cap" shape={AIButtonShape.leftCap}>
        Left Cap
      </AIButton>
      <AIButton aria-label="Right Cap" shape={AIButtonShape.rightCap}>
        Right Cap
      </AIButton>
      <AIButton aria-label="Round" shape={AIButtonShape.round} leadingIcon />
    </ButtonGroup>
  );
}
```

## Text Transform

Text can be `uppercase` (default) or `none`.

```tsx
import React from 'react';

import { AIButton, AIButtonTextTransform, ButtonGroup } from 'react-magma-dom';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton
        aria-label="Uppercase transform"
        textTransform={AIButtonTextTransform.uppercase}
      >
        Uppercase transform
      </AIButton>
      <AIButton
        aria-label="No Text Transform"
        textTransform={AIButtonTextTransform.none}
      >
        No Text Transform
      </AIButton>
    </ButtonGroup>
  );
}
```

## Full Width

Set `isFullWidth` to make the button expand to the container width.

```tsx
import React from 'react';

import { AIButton } from 'react-magma-dom';

export function Example() {
  return (
    <AIButton aria-label="Full Width" isFullWidth>

    </AIButton>
  );
}
```

## Animated Gradient

The `isAnimated` prop can be used to set gradient animation to background in the AI Button.
Because of the problems that animation can create for people who are sensitive to such things,
we automatically turn off the animation for users who have `Reduced Motion` turned on.

```tsx
import React from 'react';

import { AIButton, AIButtonVariant, ButtonGroup } from 'react-magma-dom';

export function Example() {
  return (
    <ButtonGroup>
      <AIButton aria-label="Animated Gradient" isAnimated>
        Animated Gradient
      </AIButton>
      <AIButton
        aria-label="Animated Second Gradient"
        isAnimated
        variant={AIButtonVariant.variantB}
      >
        Animated Second Gradient
      </AIButton>
      <AIButton
        aria-label="Animated Custom Gradient"
        isAnimated
        leftColor="#f00"
        rightColor="#0f0"
      >
        Animated Custom Gradient
      </AIButton>
    </ButtonGroup>
  );
}
```

## Loading State

Set `isLoading` to show a spinner and disable the button.

```tsx
import React from 'react';

import { AIButton, ButtonGroup } from 'react-magma-dom';

export function Example() {
  const [isLoading, setIsLoading] = React.useState(false);
  const [isLoadingIcon, setIsLoadingIcon] = React.useState(false);

  React.useEffect(() => {
    if (isLoading) {
      setTimeout(() => {
        setIsLoading(false);
      }, 4000);
    }

    if (isLoadingIcon) {
      setTimeout(() => {
        setIsLoadingIcon(false);
      }, 4000);
    }
  }, [isLoading, isLoadingIcon]);

  return (
    <ButtonGroup>
      <AIButton
        aria-label="Loading Button"
        isLoading={isLoading}
        onClick={() => setIsLoading(true)}
      >
        Loading Button
      </AIButton>
      <AIButton
        aria-label="Loading Icon Button"
        isLoading={isLoadingIcon}
        onClick={() => setIsLoadingIcon(true)}
      />
    </ButtonGroup>
  );
}
```

## Inverse

Use `isInverse` for dark backgrounds.

```tsx
import React from 'react';

import {
  AIButton,
  AIButtonVariant,
  Card,
  CardBody,
  ButtonGroup,
} from 'react-magma-dom';

export function Example() {
  return (
    <Card isInverse>
      <CardBody>
        <ButtonGroup>
          <AIButton aria-label="Inverse Variant A" isInverse>
            Inverse Variant A
          </AIButton>
          <AIButton
            aria-label="Inverse Variant B"
            isInverse
            variant={AIButtonVariant.variantB}
          >
            Inverse Variant B
          </AIButton>
        </ButtonGroup>
      </CardBody>
    </Card>
  );
}
```

## Event Handling

The `onClick` event will fire when the button is clicked, or when the enter or space bar key is pressed.
Events do not fire on disabled buttons.

```tsx
import React from 'react';

import { AIButton, ButtonGroup } from 'react-magma-dom';

export function Example() {
  const [counter, setCounter] = React.useState<number>(0);

  function handleUpdateCounter() {
    setCounter(count => count + 1);
  }

  return (
    <>
      <p>
        <strong>Counter: </strong> <span id="counter">{counter}</span>
      </p>
      <ButtonGroup>
        <AIButton aria-label="Increment Counter" onClick={handleUpdateCounter}>
          Click me
        </AIButton>
        <AIButton
          aria-label="Increment Counter Disabled"
          disabled
          onClick={handleUpdateCounter}
        >
          Click me
        </AIButton>
      </ButtonGroup>
    </>
  );
}
```
