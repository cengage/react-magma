## Basic Usage

Icon buttons have an icon included in them. If children are included with these buttons (usually text), then they render in a style similar to the non-icon buttons.
If no children are included, then they render in the icon-only style.

<Alert>
  For icon only buttons you <strong>must</strong> include an{' '}
  <inlineCode class="inverse">aria-label</inlineCode> property that accurately
  describes the function of the button, so that screen-reader users will know
  the button's purpose.
</Alert>
<Alert>
  For icon buttons with text, an{' '}
  <inlineCode class="inverse">aria-label</inlineCode> is not needed, as the
  function of the button should be described in the visible button text.
</Alert>

```tsx
import React from 'react';
import { IconButton, ButtonVariant, ButtonColor } from 'react-magma-dom';
import {
  AlertIcon,
  BellIcon,
  Cog2Icon,
  CheckIcon,
  InfoIcon,
} from 'react-magma-icons';

export function Example() {
  return (
    <>
      <p>
        <IconButton aria-label="Notifications" icon={<BellIcon />} />
        <IconButton aria-label="Notifications" icon={<BellIcon />} />
        <IconButton
          aria-label="Notifications"
          variant={ButtonVariant.outline}
          icon={<BellIcon />}
        />
        <IconButton
          aria-label="Notifications"
          variant={ButtonVariant.link}
          icon={<BellIcon />}
        />
        <IconButton
          aria-label="Settings"
          color={ButtonColor.secondary}
          icon={<Cog2Icon />}
        />
        <IconButton
          aria-label="Settings"
          color={ButtonColor.secondary}
          variant={ButtonVariant.outline}
          icon={<Cog2Icon />}
        />
        <IconButton
          aria-label="Settings"
          color={ButtonColor.secondary}
          variant={ButtonVariant.link}
          icon={<Cog2Icon />}
        />
        <IconButton
          aria-label="Correct"
          color={ButtonColor.success}
          icon={<CheckIcon />}
        />
        <IconButton
          aria-label="Correct"
          color={ButtonColor.success}
          variant={ButtonVariant.outline}
          icon={<CheckIcon />}
        />
        <IconButton
          aria-label="Correct"
          color={ButtonColor.success}
          variant={ButtonVariant.link}
          icon={<CheckIcon />}
        />
        <IconButton
          aria-label="Error"
          color={ButtonColor.danger}
          icon={<AlertIcon />}
        />
        <IconButton
          aria-label="Error"
          color={ButtonColor.danger}
          variant={ButtonVariant.outline}
          icon={<AlertIcon />}
        />
        <IconButton
          aria-label="Error"
          color={ButtonColor.danger}
          variant={ButtonVariant.link}
          icon={<AlertIcon />}
        />
        <IconButton aria-label="Notifications" disabled icon={<BellIcon />} />
      </p>
      <p>
        <IconButton icon={<BellIcon />}>Notifications</IconButton>
        <IconButton
          icon={<Cog2Icon />}
          color={ButtonColor.secondary}
          variant={ButtonVariant.outline}
        >
          Settings
        </IconButton>
        <IconButton
          icon={<CheckIcon />}
          color={ButtonColor.success}
          variant={ButtonVariant.link}
        >
          Submit
        </IconButton>
      </p>
    </>
  );
}
```

## Assign Icon to Variable

If you find JSX being passed in as a prop directly ugly, you can assign an icon to a variable and pass that in as the prop.

```tsx
import React from 'react';
import { IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';

export function Example() {
  const icon = <BellIcon />;

  return <IconButton aria-label="Notifications" icon={icon} />;
}
```

## Custom Size

Icon buttons have default sizes for small, medium and large icon buttons respectively. However, if a pixel size is specified for an icon itself, that size will take precedence.

```tsx
import React from 'react';
import { IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';

export function Example() {
  const icon = <BellIcon size={10} />;

  return (
    <IconButton icon={icon}>Notification with Custom Icon Size</IconButton>
  );
}
```

## Icon Position

Use the optional prop `iconPosition` to position the icon of the button on either side of the text.
The `iconPosition` property takes the values `left` and `right`.

```tsx
import React from 'react';
import { ButtonColor, ButtonIconPosition, IconButton } from 'react-magma-dom';
import { BellIcon, InfoIcon } from 'react-magma-icons';

export function Example() {
  return (
    <>
      <IconButton icon={<BellIcon />}>Icon Left</IconButton>
      <IconButton
        color={ButtonColor.secondary}
        icon={<InfoIcon />}
        iconPosition={ButtonIconPosition.right}
      >
        Icon Right
      </IconButton>
    </>
  );
}
```

## Sizes

Sizes for `IconButtons` include `small`, `medium`, and `large`, with `medium` being the default value.

```tsx
import React from 'react';
import { ButtonIconPosition, ButtonSize, IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';

export function Example() {
  return (
    <>
      <p>
        <IconButton
          icon={<BellIcon aria-label="Notifications" />}
          size={ButtonSize.small}
        />
        <IconButton icon={<BellIcon aria-label="Notifications" />} />
        <IconButton
          icon={<BellIcon aria-label="Notifications" />}
          size={ButtonSize.large}
        />
      </p>
      <p>
        <IconButton
          icon={<BellIcon />}
          size={ButtonSize.small}
          iconPosition={ButtonIconPosition.left}
        >
          Small Notifications
        </IconButton>
        <IconButton icon={<BellIcon />} iconPosition={ButtonIconPosition.left}>
          Medium Notifications
        </IconButton>
        <IconButton
          icon={<BellIcon />}
          size={ButtonSize.large}
          iconPosition={ButtonIconPosition.left}
        >
          Large Notifications
        </IconButton>
      </p>
    </>
  );
}
```

## Shape

Icon only buttons have `shape = round`, by default.

```tsx
import React from 'react';
import { ButtonShape, IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';

export function Example() {
  return (
    <>
      <IconButton icon={<BellIcon aria-label="Notifications" />} />
      <IconButton
        icon={<BellIcon aria-label="Notifications" />}
        shape={ButtonShape.fill}
      />
      <IconButton
        icon={<BellIcon aria-label="Notifications" />}
        shape={ButtonShape.rightCap}
      />
      <IconButton
        icon={<BellIcon aria-label="Notifications" />}
        shape={ButtonShape.leftCap}
      />
    </>
  );
}
```

## Full-Width Buttons

The `isFullWidth` property will set the button to display full-width. This property does not have any effect on icon-only buttons.

```tsx
import React from 'react';
import { ButtonIconPosition, IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';

export function Example() {
  return (
    <>
      <IconButton icon={<BellIcon />} isFullWidth>
        Full-Width Notifications Button
      </IconButton>
      <IconButton
        icon={<BellIcon />}
        isFullWidth
        iconPosition={ButtonIconPosition.right}
      >
        Full-Width Notifications Button
      </IconButton>

      <IconButton icon={<BellIcon aria-label="Notifications" />} isFullWidth />
    </>
  );
}
```

## Event Handling

The `onClick` event will fire when the button is clicked, or when the enter or spacebar key is pressed.
Events do not fire on disabled buttons.

```tsx
import React from 'react';
import { IconButton } from 'react-magma-dom';
import { BellIcon } from 'react-magma-icons';
export function Example() {
  const [iconButtonCounter, setIconButtonCounter] = React.useState(0);
  const [textIconButtonCounter, setTextIconButtonCounter] = React.useState(0);

  return (
    <>
      <p>
        <strong>Icon Counter: </strong> {iconButtonCounter} |{' '}
        <strong>Text Icon Counter: </strong> {textIconButtonCounter}
      </p>
      <IconButton
        icon={<BellIcon />}
        onClick={() => {
          setIconButtonCounter(count => count + 1);
        }}
      >
        Click me
      </IconButton>
      <IconButton
        icon={<BellIcon />}
        textPosition="left"
        onClick={() => {
          setTextIconButtonCounter(count => count + 1);
        }}
      >
        Click me
      </IconButton>
      <IconButton
        disabled
        icon={<BellIcon />}
        onClick={() => {
          setIconButtonCounter(count => count + 1);
        }}
      >
        Click me if you can
      </IconButton>
    </>
  );
}
```
